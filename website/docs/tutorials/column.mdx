# Column Chart

```jsx live
function showComponent() {
  const data = [
    ['Year', 'Things', 'Stuff'],
    ['2004', 1000, 400],
    ['2005', 1170, 460],
    ['2006', 660, 1120],
    ['2007', 1030, 540],
  ];
  const options = {
    categoryAxis: {
      stacked: true,
      enableColor: false,
    },
    valueAxis: {
      stacked: true,
      title: 'value',
    },
    title: 'My First Column Chart',
    theme: 'qualitative-colors-primary',
    legend: {
      isLegendClick: true,
    },
  };
  return <mdw-xy type="column" data={JSON.stringify(data)} options={JSON.stringify(options)}></mdw-xy>;
}
```
